  const handleVerify = async () => {
    try {
      const prembly = new PremblyPass({
        app_id: process.env.REACT_APP_PREMBLY_APP_ID,
        x_api_key: process.env.REACT_APP_PREMBLY_API_KEY,
        environment: 'test'
      });

      const launchResult = prembly.launch({
        config_id: process.env.REACT_APP_PREMBLY_CONFIG_ID,
        user_ref: 'user_123',
        onSuccess: () => setSuccess(true),
        callback: () => setSuccess(true)
      });

      if (launchResult instanceof Promise) {
        await launchResult;
        setSuccess(true);
      }
    } catch (error) {
      console.error("Verification failed", error);
    }
  };

  return (
    <div>
      <button id="verify-btn" onClick={handleVerify}>Verify</button>
      {success && <div id="success-message">Verification successful</div>}
    </div>
  );
};

export default FaceLiveliness;
