The function returns the `data` object from the API response. In case of an API error (non-2xx response), it returns the error data from the response.

```javascript
const axios = require('axios');

async function verifyNIN(ninNumber) {
    const url = 'https://api.prembly.com/verification/vnin';
    const appId = process.env.PREMBLY_APP_ID;
    const apiKey = process.env.PREMBLY_API_KEY;

    try {
        const response = await axios.post(url, {
            number: ninNumber
        }, {
            headers: {
                'app-id': appId,
                'x-api-key': apiKey,
                'Content-Type': 'application/json'
            }
        });

        return response.data;
    } catch (error) {
        if (error.response) {
            return error.response.data;
        }
        throw error;
    }
}

module.exports = { verifyNIN };
```
