// src/FaceLiveliness.js
import React, { useState } from 'react';
import { PremblyPass } from 'prembly-pass';

const FaceLiveliness = () => {
  const [success, setSuccess] = useState(false);

  const handleVerify = () => {
    const prembly = new PremblyPass({
      app_id: process.env.REACT_APP_PREMBLY_APP_ID,
      x_api_key: process.env.REACT_APP_PREMBLY_API_KEY,
      environment: 'test',
    });

    prembly.launch({
      config_id: process.env.REACT_APP_PREMBLY_CONFIG_ID,
      user_ref: 'user_123',
      onSuccess: () => {
        setSuccess(true);
      },
    });
  };

  return (
    <div>
      <button id="verify-btn" onClick={handleVerify}>
        Verify Identity
      </button>
      {success && <div id="success-message">Verification successful</div>}
    </div>
  );
};

export default FaceLiveliness;
