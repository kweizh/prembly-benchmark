const express = require('express');
const { createProxyMiddleware } = require('http-proxy-middleware');

const app = express();
const port = 3000;

app.use('/api/prembly', createProxyMiddleware({
  target: 'http://localhost:3001',
  changeOrigin: true,
  pathRewrite: {
    '^/api/prembly': '',
  },
  on: {
    proxyReq: (proxyReq, req, res) => {
      if (process.env.PREMBLY_APP_ID) {
        proxyReq.setHeader('app-id', process.env.PREMBLY_APP_ID);
      }
      if (process.env.PREMBLY_API_KEY) {
        proxyReq.setHeader('x-api-key', process.env.PREMBLY_API_KEY);
      }
    }
  }
}));

app.listen(port, () => {
  console.log(`Test proxy server listening at http://localhost:${port}`);
});
